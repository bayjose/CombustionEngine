set-persist=true
set-loop=true
define-boolean:shouldMove = true

//speed variable
define-int:speed = 5

init:{
set-y=256
cmd_createChannel:ships
cmd_setChannel:ships
Component_Collision{
0 -70 0
25 0 0 
35 50 0
0 -10 0
-35 50 0
-25 0 0 
}
Component_Image{
square.png
}
cmd_console:script:(this)
cmd_addMessage:Current Channel:(activeChannel) has #(activeChannelSize) bodies in it.
//cmd_loadScript:doubleFor
cmd_texture:0 SamWinters.jpg
}

move:{
if:(key a true){
	cmd_rotate:0 ccw -5
}
if:(key d true){
	cmd_rotate:0 cw 5
}
if:(key w true){
	cmd_move:0 ~(speed) ~(speed) 0
}
if:(key s true){
	cmd_move:0 ~(* speed -1) ~(* speed -1) 0
}


//the square camera movement thing, and retexture that square to a camera at some point.
if:(key up true){
	set-y=(- y speed)
}

if:(key down true){
	set-y=(+ y speed)
}
if:(key left true){
	set-x=(- x speed)
}

if:(key right true){
	set-x=(+ x speed)
}
//end of move method
}

if:(shouldMove){
	ref-method:move
}
cmd_setCam:(+ (* x -1) (halfScreenWidth)) (+ (* y -1) (halfScreenHeight)) 0 0
